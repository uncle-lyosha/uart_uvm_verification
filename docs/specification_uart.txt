//----------------------------------------------------------------------------------------
//   UART_interface
//----------------------------------------------------------------------------------------
//
//     _______      _________________________________________________________________
//            |     |     |     |     |     |     |     |     |     |     |     |
//            |     |  0  |  1  |  2  |  3  |  4  |  5  |  6  |  7  |  stop bit
//            |_____|_____|_____|_____|_____|_____|_____|_____|_____|   (1 bit)
//                  |<---------------- data [7:0] ----------------->|
//
//   Baund_Rate   = 115200;
//   Duration bit = 1 / Baund_Rate = 8.681 us;
//   Transmit data (LSB)
//
//
//----------------------------------------------------------------------------------------
//	 Transmitter
//----------------------------------------------------------------------------------------
//                   UART_tx
//            _____________________
//            |                   |
//        --->| clk     data[7:0] |<--- 
//        --->| rst               |
//            |                   |
//            |           	ready |--->
//        <---| tx          valid |<---
//            |___________________|
//
//----------------------------------------------------------------------------------------
//		Receiver
//----------------------------------------------------------------------------------------
//                   UART_rx
//            _____________________
//            |                   |
//        --->| clk     data[7:0] |---> 
//        --->| rst               |
//            |                   |
//            |           tx_done |--->
//        --->| rx                |
//            |___________________|
//
